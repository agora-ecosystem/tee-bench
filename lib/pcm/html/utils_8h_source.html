167  {
168  istr >> std::noskipws;
169  const auto len = length();
170  char * buffer = new char[len + 2];
171  buffer[0] = 0;
172  istr.get(buffer, len+1);
173  if (*this != std::string(buffer))

175  istr.setstate(std::ios_base::failbit);

177  delete [] buffer;
178  }
