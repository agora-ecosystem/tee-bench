69  template<typename T>
70  void dyn_hex_table_output( int debugLevel, std::ostream& out, ssize_t len, T* inputBuffer_ ) {
71  std::stringstream ss;
72  if ( debug::currentDebugLevel < debugLevel )
73  return;
74  for ( ssize_t i = 0; i < len; ++i ) {
75  constexpr int DHTO_CHARS_PER_LINE = 16;
76  ss << std::hex << std::internal << std::setfill('0') << std::setw(2) << std::abs(inputBuffer_[i]) << " ";
77  if ( (i % DHTO_CHARS_PER_LINE) == (DHTO_CHARS_PER_LINE - 1) ) ss << "\n";
78  }
79  out << ss.str() << std::flush;
80  }
