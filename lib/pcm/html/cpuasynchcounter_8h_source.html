187 void * UpdateCounters(void * state)
188 {
189  AsynchronCounterState * s = (AsynchronCounterState *)state;
190 
191  while (true) {
192  pthread_mutex_lock(&(s->CounterMutex));
193  for (uint32 core = 0; core < s->m->getNumCores(); ++core) {
194  s->cstates1[core] = std::move(s->cstates2[core]);
195  s->cstates2[core] = std::move(s->m->getCoreCounterState(core));
196  }
197 
198  for (uint32 socket = 0; socket < s->m->getNumSockets(); ++socket) {
199  s->skstates1[socket] = std::move(s->skstates2[socket]);
200  s->skstates2[socket] = std::move(s->m->getSocketCounterState(socket));
201  }
202 
203  s->sstate1 = std::move(s->sstate2);
204  s->sstate2 = std::move(s->m->getSystemCounterState());
205 
206  pthread_mutex_unlock(&(s->CounterMutex));
207  sleep(1);
208  }
209  return NULL;
210 }
