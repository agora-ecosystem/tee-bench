146 void probe_pci(struct pci *p)
147 {
148  uint32 value;
149  struct bdf *bdf = &p->bdf;
150  if (PciHandleType::exists(0, bdf->busno, bdf->devno, bdf->funcno)) {
151  p->exist = true;
152  PciHandleType h(0, bdf->busno, bdf->devno, bdf->funcno);
153  h.read32(0x0, &value); //VID:DID
154  if (value == (std::numeric_limits<unsigned int>::max)()) // invalid VID::DID
155  {
156  p->exist = false;
157  return;
158  }
159  p->offset_0 = value;
160  h.read32(0xc, &value);
161  p->header_type = (value >> 16) & 0x7f;
162  if (p->header_type == 0) {
163  h.read32(0x4, &value); //Status register
164  if (value & 0x100000) {//Capability list == true
165  h.read32(0x34, &value); //Capability pointer
166  probe_capability_pci_express(p, value);
167  }
168  } else if (p->header_type == 1) {
169  h.read32(0x18, &value);
170  p->offset_18 = value;
171  }
172  }
173  else
174  p->exist = false;
175 }
